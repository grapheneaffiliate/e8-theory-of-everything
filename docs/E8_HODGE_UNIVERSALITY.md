def generate_e8_roots():

    # Type 1: D8 roots
    for i in range(8):
        for j in range(i+1, 8):
            for s1, s2 in [(1,1), (1,-1), (-1,1), (-1,-1)]:
                v = np.zeros(8)
                v[i], v[j] = s1, s2
                roots.add(tuple(v))
    
    # Type 2: Spinor roots
    for mask in range(256):
        signs = [(2*(mask>>i & 1)-1) for i in range(8)]
        if sum(s == -1 for s in signs) % 2 == 0:
            roots.add(tuple(np.array(signs)/2))
    
    return list(roots)
